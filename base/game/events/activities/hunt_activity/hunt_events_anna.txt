#######################
#	Carving initials of your loved one into the bark of a tree
#	by Anna Johansson
#######################

hunt_events_anna.0001 = {
	type = activity_event
	title = hunt_events_anna.0001.t
	desc = hunt_events_anna.0001.desc
	theme = hunt_activity
	left_portrait = {
		character = root
		animation = love
	}
	lower_left_portrait = {
		character = scope:love_interest
	}

	cooldown = { years = 40 }
	
	trigger = {
		OR = {
			any_relation = {
				type = lover
			}
			any_relation = {
				type = soulmate
			}
		}
	}

	immediate = {
		if = {
			limit = {
				any_relation = {
					type = soulmate
				}
			}
			random_relation = {
				type = soulmate
				save_scope_as = love_interest
			}
		}
		else = {
			random_relation = {
				type = lover
				save_scope_as = love_interest
			}
		}
	}

	# Carve names into the bark
	option = { 
		name = hunt_events_anna.0001.a

		add_stress = major_stress_loss

		add_character_modifier = {
			modifier = hunt_carving_into_bark_modifier
			years = 5
		}

		stress_impact = {
			impatient = minor_stress_impact_gain
			compassionate = minor_stress_impact_loss
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_trait = impatient
			}
			modifier = {
				factor = 2
				has_trait = compassionate
			}
		}
	}

	# No time to waste
	option = {
		name = hunt_events_anna.0001.b
		random = {
			chance = 70
			send_interface_toast = {
				type = event_toast_effect_good
				title = hunt_events_anna.0001.b.t_toast
				left_icon = root
				hunt_activity_success_change_effect = { CHANGE = increase_minor }
			}
		}

		stress_impact = {
			impatient = minor_stress_impact_loss
		}

		ai_chance = {
			base = 100
		}
	}
}

hunt_events_anna.0005 = {
	type = activity_event
	title = hunt_events_anna.0005.t
	desc = hunt_events_anna.0005.desc
	theme = hunt_activity
	left_portrait = {
		character = root
		animation = personality_cynical
	}
	right_portrait = {
		character = scope:mocking_participant
		animation = hunting_carcass_start
	}

	cooldown = { years = 15 }
	
	trigger = {
		involved_activity = {
			any_attending_character = {
				is_adult = yes
				is_ai = yes
				NOR = {
					this = root
					is_close_family_of = root
					is_spouse_of = root
					is_concubine_of = root
				}
				OR = {
					opinion = {
						target = root
						value < 0
					}
					has_relation_rival = root
				}
			}
		}
	}

	immediate = {
		involved_activity = {
			random_attending_character = {
				limit = {
					is_adult = yes
					is_ai = yes
					NOR = {
						this = root
						is_close_family_of = root
						is_spouse_of = root
						is_concubine_of = root
					}
					OR = {
						opinion = {
							target = root
							value < 0
						}
						has_relation_rival = root
					}
				}
				save_scope_as = mocking_participant
			}
		}
	}

	# Flay the animal
	option = { 
		name = hunt_events_anna.0005.a

		duel = {
			skill = learning
			value = medium_skill_rating
			# Success
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				send_interface_toast = {
					type = event_toast_effect_good
					title = hunt_events_anna.0005.a.toast.good
					left_icon = root
					if = {
						limit = {
							has_trait = lifestyle_hunter
						}
						add_trait_xp = {
							trait = lifestyle_hunter
							track = hunter
							value = minor_trait_xp
						}
					}
					else = {
						add_learning_lifestyle_xp = medium_lifestyle_xp
						random = {
							chance = 10
							add_trait = lifestyle_hunter
						}
					}
				}
			}
			# Failure
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				send_interface_toast = {
					type = event_toast_effect_neutral
					title = hunt_events_anna.0005.a.toast.nothing
					left_icon = root
					add_prestige = minor_prestige_loss
				}				
			}
		}

		stress_impact = {
			compassionate = minor_stress_impact_gain
			diligent = minor_stress_impact_loss
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_trait = compassionate
			}
			modifier = {
				factor = 2
				has_trait = diligent
			}
		}
	}

	# Flay the person instead
	option = {
		name = hunt_events_anna.0005.b

		trigger = { has_trait = sadistic }
		trait = sadistic
		scope:mocking_participant = {
			death = {
				killer = root
				death_reason = death_hunting_mysterious
			}
		}
		add_dread = medium_dread_gain

		stress_impact = {
			sadistic = medium_stress_impact_loss
		}

		add_secret = {
			type = secret_murder
			target = scope:mocking_participant
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_trait = compassionate
			}
			modifier = {
				factor = 2
				has_trait = sadistic
			}
		}
	}

	# Do not engage
	option = {
		name = hunt_events_anna.0005.c

		reverse_add_opinion = {
			target = scope:mocking_participant
			modifier = disrespect_opinion
			opinion = -20
		}

		stress_impact = {
			sadistic = minor_stress_impact_gain
			compassionate = minor_stress_impact_loss
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_trait = sadistic
			}
			modifier = {
				factor = 2
				has_trait = compassionate
			}
		}
	}
}

hunt_events_anna.0010 = {
	type = activity_event
	title = hunt_events_anna.0010.t
	desc = hunt_events_anna.0010.desc
	theme = hunt_activity
	left_portrait = {
		character = root
		animation = severelywounded
	}

	cooldown = { years = 40 }

	trigger = {
		involved_activity = {
			any_attending_character = {
				is_adult = yes
				this != root
			}
		}
	}

	immediate = {
		increase_wounds_effect = { REASON = fall }
	}

	# Find a way to get out
	option = {
		name = hunt_events_anna.0010.a

		duel = {
			skill = prowess
			value = medium_skill_rating
			# Success
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				desc = hunt_events_anna.0010.a.desc.good
				send_interface_toast = {
					type = event_toast_effect_good
					title = hunt_events_anna.0010.a.toast.good
					left_icon = root
					add_prowess_skill = 1
					add_prestige = medium_prestige_gain
				}
			}

			# Failure
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				desc = hunt_events_anna.0010.a.desc.bad
				send_interface_toast = {
					type = event_toast_effect_neutral
					title = hunt_events_anna.0010.a.toast.bad
					left_icon = root
					increase_wounds_effect = { REASON = fall }
				}				
			}
		}

		stress_impact = {
			craven = minor_stress_impact_gain
			lazy = minor_stress_impact_gain
			brave = minor_stress_impact_loss
			diligent = minor_stress_impact_loss
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				OR = {
					has_trait = craven
					has_trait = lazy
				}
			}
			modifier = {
				factor = 2
				OR = {
					has_trait = brave
					has_trait = diligent
				}
			}
		}
	}

	# Scream for help
	option = {
		name = hunt_events_anna.0010.b
		flavor = hunt_events_anna.0010.b.flavor

		hunt_activity_success_change_effect = { CHANGE = decrease_minor }

		stress_impact = {
			brave = minor_stress_impact_gain
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_trait = brave
			}
			modifier = {
				factor = 2
				has_trait = craven
			}
		}
	}
}

#######################
#	Helping your child to become a better shot
#	by Anna Johansson
#######################

hunt_events_anna.0015 = {
	type = activity_event
	title = hunt_events_anna.0015.t
	desc = hunt_events_anna.0015.desc
	theme = hunt_activity
	left_portrait = {
		character = root
		animation = frontend_center_idle
	}
	right_portrait = {
		character = scope:child_with_bow
		scripted_animation = bow_drawn
	}

	cooldown = { years = 30 }

	trigger = {
		has_trait = lifestyle_hunter
		involved_activity = {
			any_attending_character = {
				any_parent = {
					this = root
				}
				NOR = {
					has_trait = lifestyle_hunter
					has_trait = tourney_participant
				}
				OR = {
					prowess < 5
					martial < 5
				}
			}
		}
	}

	immediate = {
		involved_activity = {
			random_attending_character = {
				limit = {
					any_parent = {
						this = root
					}
					NOR = {
						has_trait = lifestyle_hunter
						has_trait = tourney_participant
					}
					OR = {
						prowess < 5
						martial < 5
					}
				}
				save_scope_as = child_with_bow
			}
		}
	}

	# Help the child out how to handle a bow
	option = { 
		name = hunt_events_anna.0015.a
		flavor = hunt_events_anna.0015.a.flavor

		random = {
			chance = {
				value = 20
				if = {
					limit = {
						has_trait_xp = {
							trait = lifestyle_hunter
							track = hunter
							value >= 80
						}
					}
					add = 80
				}
				else_if = {
					limit = {
						has_trait_xp = {
							trait = lifestyle_hunter
							track = hunter
							value >= 60
						}
					}
					add = 60
				}
				else_if = {
					limit = {
						has_trait_xp = {
							trait = lifestyle_hunter
							track = hunter
							value >= 40
						}
					}
					add = 40
				}
				else_if = {
					limit = {
						has_trait_xp = {
							trait = lifestyle_hunter
							track = hunter
							value >= 20
						}
					}
					add = 20
				}
			}

			send_interface_toast = {
				type = event_toast_effect_good
				title = hunt_events_anna.0015.a.toast.good
				left_icon = root
				right_icon = scope:child_with_bow
				scope:child_with_bow = {
					if = {
						limit = {
							NOT = { has_trait = lifestyle_hunter }
						}
						add_trait = lifestyle_hunter
					}
				}
			}
		}

		stress_impact = {
			impatient = minor_stress_impact_gain
			patient = minor_stress_impact_loss
			compassionate = minor_stress_impact_loss
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_trait = impatient
			}
			modifier = {
				factor = 2
				OR = {
					has_trait = patient
					has_trait = compassionate

				}
			}
		}
	}

	# Cheer the child on
	option = {
		name = hunt_events_anna.0015.b
		progress_towards_friend_effect = {
			REASON = friend_supported_child
			CHARACTER = scope:child_with_bow
			OPINION = default_friend_opinion
		}

		stress_impact = {
			impatient = minor_stress_impact_gain
			compassionate = minor_stress_impact_loss
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_trait = impatient
			}
			modifier = {
				factor = 2
				has_trait = compassionate
			}
		}
	}
	
	# Observe the child from a distance
	option = {
		name = hunt_events_anna.0015.c

		random = {
			chance = 10
			send_interface_toast = {
				title = hunt_events_anna.0015.c.toast
				left_icon = scope:child_with_bow
				scope:child_with_bow = {
					if = {
						limit = {
							NOT = { has_trait = lifestyle_hunter }
						}
						add_trait = lifestyle_hunter
					}
				}
			}
		}

		stress_impact = {
			ambitious = minor_stress_impact_gain
			content = minor_stress_impact_loss
			compassionate = minor_stress_impact_loss
		}

		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_trait = ambitious
			}
			modifier = {
				factor = 2
				OR = {
					has_trait = content
					has_trait = compassionate
				}
			}
		}
	}
}
