# travel_events.5100 - Reaching the Troll Bridge

travel_events.5100 = { 
    type = character_event
    title = travel_events.5100.t
    desc = travel_events.5100.desc
    theme = travel
    left_portrait = {
		character = root
		animation = survey
	}
    right_portrait = {
        character = scope:superstitious_traveler
        animation = paranoia
        camera = camera_event_kneeling
    }
    override_background = ep2_travel_bridge
    override_effect_2d = { reference = fog }
    widget = { 
    	gui = event_window_widget_vfx_fog 
    	container = foreground_shader_vfx_container 
    }
    cooldown = { years = 10 }
	
    trigger = {
    	static_group_filter = {
    		group = travel_events.5100
    		match = 0.5
    	}
    	is_location_valid_for_travel_event_on_land = yes
    	current_travel_plan = {
			any_entourage_character = {
				NOT = { has_trait = cynical }
				this != root
				age >= 7
			}
		}
		location = { is_drylands_or_desert_trigger = no }
    }
	
    immediate = {
    	current_travel_plan = {
			random_entourage_character = {
				limit = {
					NOT = { has_trait = cynical }
					this != root
					age >= 7
				}
				save_scope_as = superstitious_traveler
			}
		}
    	
	}
    #Brave - get some positive opinion and poistive travel modifier
	option = { 
        name = travel_events.5100.a
		trigger = {
			has_trait = brave
		}
		current_travel_plan = {
			add_travel_plan_modifier = intrepid_traveler_travel_modifier
			every_entourage_character = {
				limit = { 
					this != root 
					OR = {
						has_trait = brave
						has_trait = impatient
					}

				}
				custom = custom.every_brave_or_impatient_entourage_member
				add_opinion = {
					target = root
					modifier = impressed_opinion
					opinion = 10
				}
			}
		}
		stress_impact = {
			brave = medium_stress_impact_loss
		}
		ai_chance = { base = 500 }
    } 
   #Insult the person that gave the warning for prestig - they lose opinion 
	option = { 
        name = travel_events.5100.b
        add_prestige = minor_prestige_gain
		scope:superstitious_traveler = {
			add_opinion = {
					target = root
					modifier = annoyed_opinion
					opinion = -10
				}
		}
		current_travel_plan = {
			every_entourage_character = {
				limit = { 
					this != root 
					this != scope:superstitious_traveler 
					OR = {
						has_trait = cynical
						has_trait = impatient
					}

				}
				custom = custom.every_cynical_or_impatient_entourage_member
				add_opinion = {
					target = root
					modifier = impressed_opinion
					opinion = 10
				}
			}
		}
		stress_impact = {
			calm = minor_stress_impact_loss
			cynical = medium_stress_impact_loss
			zealous = minor_stress_impact_gain
		}	
		ai_chance = { base = 100 }
    } 
	#Pray to make the troll go away, gain piety but lose some time
	option = { 
        name = travel_events.5100.c
        add_piety = minor_piety_gain
        current_travel_plan = {
        delay_travel_plan = { days = 3 }
			every_entourage_character = {
				limit = { 
					this != root 
					OR = {
						has_trait = eccentric
						has_trait = zealous
					}

				}
				custom = custom.every_eccentric_or_zealous_entourage_member
				add_opinion = {
					target = root
					modifier = impressed_opinion
					opinion = 10
				}
			}
		}
		stress_impact = {
			eccentric = minor_stress_impact_loss
			zealous = medium_stress_impact_loss
			impatient = medium_stress_impact_gain
		}	

		ai_chance = { base = 100 }
    } 
    #Craven - lose time but gain travel experience
    option = { 
        name = travel_events.5100.d
        trigger = {
			has_trait = craven
		}
		if = {
			limit = {
				NOT = { has_trait = lifestyle_traveler }
			}
			add_trait = lifestyle_traveler
		}
		else_if = {
			limit = {
				has_trait = lifestyle_traveler
				has_trait_xp = {
					trait = lifestyle_traveler
					track = travel
					value < 100
				}
			}
			add_trait_xp = {
				trait = lifestyle_traveler
				track = travel
				value = { 5 10 }
			}
		}
        current_travel_plan = {
        	delay_travel_plan = { days = 7 }
			every_entourage_character = {
				limit = { 
					this != root 
					OR = {
						has_trait = craven
						has_trait = impatient
					}

				}
				custom = custom.every_craven_or_impatient_entourage_member
				add_opinion = {
					target = root
					modifier = impressed_opinion
					opinion = 10
				}
			}
		}
		stress_impact = {
			craven = medium_stress_impact_loss
		}
		ai_chance = { base = 500 }
    }                              	
}
