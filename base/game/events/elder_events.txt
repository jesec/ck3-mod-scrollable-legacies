namespace = elder_events

scripted_trigger suitable_elder = {
	is_physically_able_adult = yes
	has_trait = withering_mind
	is_close_family_of = root
	NOR = {
		is_sibling_of = root
		has_relation_rival = root
	}
}

# Withering Mind - Relative forgets your name
elder_events.1000 = {
	type = character_event
	title = elder_events.1000.t
	desc = elder_events.1000.desc
	theme = mental_health
	
	override_background = { reference = relaxing_room }

	left_portrait = {
		character = root
		animation = sadness
	}

	right_portrait = {
		character = scope:elder
		animation = admiration
		camera = camera_event_center_pointing_left
	}
	
	lower_right_portrait = scope:confusion_target
	
	trigger = {
		is_available = yes
		has_trait_malicious_trigger = no
		any_courtier_or_guest = {
			suitable_elder = yes
		}
		any_sibling = {
			even_if_dead = yes
			age <= age_plus_10
			age >= age_minus_10
			sex_same_as = root
		}
		NOT = {
			has_trait = withering_mind
		}
	}

	cooldown = { years = 25 }

	immediate = {
		save_scope_as = root_scope
		random_courtier_or_guest = {
			limit = {
				suitable_elder = yes
			}
			save_scope_as = elder
		}
		random_sibling = {
			even_if_dead = yes
			limit = {
				age <= age_plus_10
				age >= age_minus_10
				sex_same_as = root
			}
			save_scope_as = confusion_target
		}
	}

	option = {
		name = elder_events.1000.a
		
		scope:elder = {
			add_stress = medium_stress_impact_gain
		}
		
		stress_impact = {
			base = minor_stress_impact_gain
			honest = minor_stress_impact_loss
		}

		ai_chance = {
			base = 100
		}
	}

	option = {
		name = elder_events.1000.b
		
		scope:elder = {
			add_trait_xp = { trait = withering_mind track = withering_mind value = 10 }
		}
		
		stress_impact = {
			base = minor_stress_impact_gain
			honest = medium_stress_impact_gain
		}

		ai_chance = {
			base = 100
		}
	}
}

# Fragile Bones - Trip and fall down the stairs
elder_events.1100 = {
	type = character_event
	title = elder_events.1100.t
	desc = elder_events.1100.desc
	theme = physical_health
	
	override_background = corridor_day

	left_portrait = {
		character = root
		animation = pain
		camera = camera_event_right_pointing_right
	}
	
	trigger = {
		has_trait = fragile_bones
		prowess < 10
		is_landed_or_landless_administrative = yes
		is_available_adult = yes
		NOT = {
			has_trait = athletic
		}
		location = {
			NOR = {
				has_holding_type = nomad_holding
				has_holding_type = herder_holding
				has_holding_type = tribal_holding
				is_nomadic_location_trigger = yes
			}
		}
	}

	cooldown = { years = 25 }

	immediate = {
		save_scope_as = root_scope
		faith = {
			random_holy_site = {
				limit = {
					title_province = {
						OR = {
							terrain = desert_mountains
							terrain = mountains
						}
					}
				}
				save_scope_as = mountain_reference
			}
		}
		if = {
			limit = {
				NOT = {
					exists = scope:mountain_reference
				}
			}
			random_realm_county = {
				limit = {
					any_county_province = {
						OR = {
							terrain = desert_mountains
							terrain = mountains
						}
					}
				}
				random_county_province = {
					limit = {
						OR = {
							terrain = desert_mountains
							terrain = mountains
						}
					}
					save_scope_as = mountain_reference
				}
			}
		}
		if = {
			limit = {
				is_ai = no
				NOT = {
					exists = scope:mountain_reference
				}
			}
			random_county = {
				limit = {
					any_county_province = {
						OR = {
							terrain = desert_mountains
							terrain = mountains
						}
						squared_distance = {
							target = root.location
							value <= squared_distance_medium
						}
					}
				}
				random_county_province = {
					limit = {
						OR = {
							terrain = desert_mountains
							terrain = mountains
						}
						squared_distance = {
							target = root.location
							value <= squared_distance_medium
						}
					}
					save_scope_as = mountain_reference
				}
			}
		}
		if = {
			limit = {
				is_ai = no
				NOT = {
					exists = scope:mountain_reference
				}
			}
			random_county = {
				limit = {
					any_county_province = {
						OR = {
							terrain = desert_mountains
							terrain = mountains
						}
					}
				}
				random_county_province = {
					limit = {
						OR = {
							terrain = desert_mountains
							terrain = mountains
						}
					}
					save_scope_as = mountain_reference
				}
			}
		}
	}

	option = {
		name = elder_events.1100.bodyguard
		
		trigger = {
			employs_court_position = bodyguard_court_position
		}
		show_as_unavailable = {
			NOT = {
				employs_court_position = bodyguard_court_position
			}
		}
		reason = has_bodyguard
		add_trait_xp = { trait = fragile_bones track = fragile_bones value = 1 }

		ai_chance = {
			base = 100
		}
	}

	option = {
		name = elder_events.1100.a
		
		trigger = {
			has_trait = fragile_bones
		}
		increase_wounds_no_death_effect = { REASON = accident }
		add_trait_xp = { trait = fragile_bones track = fragile_bones value = 3 }

		ai_chance = {
			base = 0
		}
	}
}

		ai_chance = {
			base = 0
		}
	}

scripted_trigger appropriate_lover = {
	is_available_ai_adult = yes
	age <= 40
	OR = {
		age <= 20
		has_trait = lustful
		has_trait = athletic
		has_trait = physique_good
		has_trait = lifestyle_reveler
		has_trait = fecund
		has_trait = seducer
		faith = {
			has_doctrine = tenet_carnal_exaltation
		}
		AND = {
			has_relation_lover = root
			root.age >= 60
		}
	}
	NOR = {
		has_trait = chaste
		has_trait = celibate
	}
}

# Faltering heart - Chance of death from sex
elder_events.1400 = {
	type = character_event
	title = elder_events.1400.t
	desc = elder_events.1400.desc
	theme = physical_health
	
	override_background = { reference = bedchamber }

	left_portrait = {
		character = root
		animation = holding_crutch
		camera = camera_event_right_pointing_left
	}

	right_portrait = {
		character = scope:lover
		animation = flirtation_left
		camera = camera_event_right_crowner_crowning
	}
	
	trigger = {
		is_available_adult = yes
		has_trait = faltering_heart
		age >= 50
		NOR = {
			has_trait = chaste
			has_trait = celibate
		}
		OR = {
			any_consort = {
				location = root.location
				is_attracted_to_gender_of = root
				appropriate_lover = yes
			}
			any_relation = {
				type = lover
				appropriate_lover = yes
			}
		}
	}

	cooldown = { years = 25 }

	immediate = {
		save_scope_as = root_scope
		add_character_flag = is_naked
		random_relation = {
			type = lover
			limit = {
				appropriate_lover = yes
			}
			save_scope_as = lover
			add_character_flag = is_naked
		}
		if = {
			limit = {
				NOT = {
					exists = scope:lover
				}
			}
			random_consort = {
				limit = {
					location = root.location
					is_attracted_to_gender_of = root
					appropriate_lover = yes
				}
				save_scope_as = lover
				add_character_flag = is_naked
			}
		}
	}

	option = {
		name = elder_events.1400.a
		
		trigger = {
			has_trait = faltering_heart
		}
		
		add_stress = {
			value = massive_stress_loss
			add = massive_stress_loss
		}
		
		had_sex_with_effect = {
			CHARACTER = scope:lover
			PREGNANCY_CHANCE = pregnancy_chance
		}
		
		random = {
			chance = {
				value = root.age
				add = -40
				min = 1
			}
			death = { death_reason = death_best_sex_of_their_life }
		}

		ai_chance = {
			base = 100
		}
	}

	option = {
		name = elder_events.1400.b
		
		add_stress = major_stress_gain
		scope:lover	= {
			add_opinion = {
				target = root
				modifier = disappointed_opinion
				opinion = -20
			}
		}

		ai_chance = {
			base = 0
		}
	}
	
	after = {
		if = {
			limit = {
				is_alive = yes
			}
			remove_character_flag = is_naked
		}
		scope:lover = {
			if = {
				limit = {
					is_alive = yes
				}
				remove_character_flag = is_naked
			}
		}
	}
}
