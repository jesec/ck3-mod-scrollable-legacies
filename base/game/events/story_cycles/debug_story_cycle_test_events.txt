namespace = debug_story_test_event

#Used to find someone who hates you
scripted_trigger debug_story_test_hate_trigger = {
	OR = {
		has_relation_rival = root
		has_relation_nemesis = root
		opinion = {
			target = root
			value <= -50
		}
	}
}
#activates story cycle
scripted_effect create_debug_story_test = {
	create_story = {
		type = debug_test_story
		save_scope_as = new_story
	}
	random_owned_story = {
		type = debug_test_story
		save_scope_as = debug_test_story
		set_variable = {
			name = debug_test_story_enemy
			value = flag:$ENEMY$
		}
		set_variable = {
			name = debug_test_story_liege
			value = scope:liege
		}
		set_variable = {
			name = debug_test_story_counter
			value = 0
		}
		set_variable = {
			name = debug_test_story_test_title
			value = scope:test_title 
		}
		set_variable = {
			name = debug_test_story_test_artifact
			value = scope:double_moon_book
		}

		story_owner = {
			every_courtier = {
				scope:debug_test_story = {
					add_to_variable_list = {
						name = debug_test_story_courtiers
						target = prev
					}
				}
			}
			liege.primary_title = {
				every_direct_de_facto_vassal_title = {
					scope:debug_test_story = {
						add_to_variable_list = {
							name = debug_test_story_title_list
							target = prev
						}
					}
				}
			}
			every_character_artifact = {
				scope:debug_test_story = {
					add_to_variable_list = {
						name = debug_test_story_artifact_list
						target = prev
					}
				}
			}
		}	
	}
}

debug_story_test_event.0001 = {
	type = character_event
	window = big_event_window
	title = debug_test_story.0001.t
	desc = debug_test_story.0001.desc

	theme = faith

	left_portrait = {
		character = root
		animation = wailing
	}
	right_portrait = {
		character = scope:liege
		animation = personality_callous
	}

	lower_right_portrait = {
		trigger = { exists = scope:bad_courtier }
		character = scope:bad_courtier
	}

	immediate = {
		create_artifact = {
			name = artifact_double_moon_book
			description = artifact_double_moon_book_description
			type = journal
			visuals = book
			modifier = double_moon_modifier
			save_scope_as = double_moon_book
		}
		random_dummy_gender_effect = yes
		create_artifact_weapon_effect = {
			OWNER = root
			CREATOR = scope:dummy_gender
			SET_WEAPON_TYPE = flag:no
		}
		random_dummy_gender_effect = yes
		create_artifact_armor_effect = {
			OWNER = root
			CREATOR = scope:dummy_gender
			SET_ARMOR_TYPE = flag:no
		}
		create_artifact_regalia_effect = {
			OWNER = root
			SMITH = root
		}
		random_dummy_gender_effect = yes
		create_artifact_crown_effect = {
			OWNER = root
			SMITH = scope:dummy_gender
		}
		random_dummy_gender_effect = yes
		create_artifact_brooch_effect = {
			OWNER = root
			SMITH = scope:dummy_gender
		}
		location.county.holder.top_liege = { save_scope_as = liege }

		scope:liege.primary_title = {
			save_scope_as = test_title
		}

		if = { #If a random courtier hates you
			limit = {
				scope:liege = {
					any_courtier = {
						location = scope:liege.location
						debug_story_test_hate_trigger = yes
					}
				}
			}
			scope:liege = {
				random_courtier = {
					limit = {
						debug_story_test_hate_trigger = yes
					}
					weight = {
						#Choose someone that really hates you first
						modifier = {
							add = 15
							has_relation_nemesis = root
						}
						modifier = {
							add = 10
							has_relation_rival = root
						}
						modifier = {
							add = 5
							opinion = {
								target = root
								value <= -90
							}
						}
						#Try to make it someone relevant
						modifier = {
							add = 5
							is_councillor_of = scope:liege
						}
						modifier = {
							add = 5
							OR = {
								is_close_family_of = scope:liege
								is_consort_of = scope:liege
							}
						}
						modifier = {
							add = 5
							is_powerful_vassal_of = scope:liege
						}
						modifier = {
							add = 2
							has_trait = callous
						}
						modifier = {
							add = 2
							has_trait = disloyal
						}
						modifier = {
							add = 2
							has_trait = ambitious
						}
					}
					save_scope_as = bad_courtier
				}
			}
			hidden_effect = {
				create_debug_story_test = { ENEMY = courtier }
			}
		}
		else = { #We make someone hate you
			scope:liege = {
				random_courtier = {
					limit = {
						is_available_healthy_ai_adult = yes
					}
					weight = {
						#Try to make it someone relevant
						modifier = {
							add = 5
							is_councillor_of = scope:liege
						}
						modifier = {
							add = 5
							OR = {
								is_close_family_of = scope:liege
								is_consort_of = scope:liege
							}
						}
						modifier = {
							add = 5
							is_powerful_vassal_of = scope:liege
						}
						modifier = {
							add = 2
							has_trait = callous
						}
						modifier = {
							add = 2
							has_trait = disloyal
						}
						modifier = {
							add = 2
							has_trait = ambitious
						}
					}
					save_scope_as = bad_courtier
				}
			}
			hidden_effect = {
				scope:bad_courtier = {
					add_opinion = {
						target = root
						opinion = -75
						modifier = hate_opinion
					}
				}
				create_debug_story_test = { ENEMY = courtier }
			}
		}
		random_owned_story = {
			type = debug_test_story
			save_scope_as = debug_test_story
		}
	}

	#Option A) Make counter go up
	option = {
		name = debug_test_story.0001.a
		flavor = debug_test_story.0001.a.flavor
		custom_tooltip = debug_test_story_counter_positive
		scope:debug_test_story = {
			change_variable = {
				name = debug_test_story_counter
				add = 5
			}
		}
		
		ai_chance = {
			base = 1000 # Historically determined
		}
		add_character_modifier = {
			modifier = debug_test_story_prayer_modifier
			years = 10
		}
		add_character_modifier = {
			modifier = debug_test_story_weak_stomach_modifier
			years = 10
		}
	}

	#Option B) Make counter go down
	option = {
		name = debug_test_story.0001.b
		flavor = debug_test_story.0001.b.flavor
		custom_tooltip = debug_test_story_counter_negative
		scope:debug_test_story = {
			change_variable = {
				name = debug_test_story_counter
				add = -5
			}
		}
		ai_chance = {
			base = 0 # Historically determined
		}
	}
	
	#Option C) END
	option = {
		name = debug_test_story.0001.c
		scope:debug_test_story = { end_story = yes }
		ai_chance = {
			base = 0 # Historically determined
		}
	}
	
}
