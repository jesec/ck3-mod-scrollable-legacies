#On actions related to health events


yearly_health_pulse = { #Fired from the random_yearly_everyone_pulse
	on_actions = {
		#disease_outbreak_pulse # moved to a 5-year pulse in yearly_on_actions.txt
		# Provides 4 opportunities per year for you to recover from being wounded
		delay = { days = { 1 89 } }
		wounded_recovery_pulse
		delay = { days = { 90 179 } }
		wounded_recovery_pulse
		delay = { days = { 180 269 } }
		wounded_recovery_pulse
		delay = { days = { 270 365 } }
		wounded_recovery_pulse
	}
	random_events = {
		# Balanced to keep the events:
		## Present but not constant on default values.
		## Rare but not never-seen on reduced values.
		## Increased without being ludicrous on high values.
		### Reduced down drastically per community feedback.
		
		chance_of_no_event = {
			if = {
				limit = {
					age <= 50
					NOT = {
						has_trait_with_flag = age_related_ailment
					}
				}
				value = 75
			}
			else = { value = 25 }
		}
		600 = 0
		
		#Infirm 
		30 = health.7200	# Your mind starts withering
		20 = health.7300	# Your eyes start to cloud
		30 = health.7400	# Your heart starts faltering
		30 = health.7500	# Your bones start weakening
		20 = health.7000	# Become infirm
		20 = health.7100	# Become depressed while infirm
		40 = harm.0001		# Becoming incapable due to age: culmination.
		40 = harm.0002		# Becoming incapable due to age: mental problems.
		40 = harm.0003		# Becoming incapable due to age: physical problems.

		# Random death & incapability
		15 = harm.9501		# Try to fire a harm event.
	}
	effect = {
		if = {
			limit = {
				has_trait_with_flag = age_related_ailment
				is_incapable = no
			}
			if = {
				limit = {
					has_trait = infirm
				}
				random_list = {
					25 = {
						modifier = {
							NOT = {
								has_trait = athletic
							}
							add = 25
						}
						modifier = {
							age >= 50
							add = 25
						}
						modifier = {
							health >= fine_health
							add = -25
						}
						send_interface_message = {
							type = msg_health_change
							title = your_body_weakens_feed_message_title
							left_icon = root
							add_trait_xp = { trait = infirm track = infirm value = 8 }
						}
					}
					60 = {
						modifier = { # The lower your prowess, the lower the chance of the 'best' result, working with a base of 15
							add = prowess
						}
						send_interface_message = {
							type = msg_health_change
							title = your_body_weakens_feed_message_title
							left_icon = root
							add_trait_xp = { trait = infirm track = infirm value = 4 }
						}
					}
					0 = {
						modifier = {
							has_any_debilitating_illness = yes
							add = 300
						}
						modifier = {
							age >= 65
							add = 50
						}
						modifier = {
							health >= fine_health
							add = -25
						}
						send_interface_message = {
							type = msg_health_change
							title = your_body_weakens_fast_feed_message_title
							left_icon = root
							add_trait_xp = { trait = infirm track = infirm value = 12 }
						}
					}
				}
			}
			if = {
				limit = {
					has_trait = withering_mind
				}
				random_list = {
					1 = {
						modifier = {
							num_of_relation_friend <= 0
							num_of_relation_lover <= 0
							num_of_relation_ward <= 0
							add = 4
						}
						modifier = {
							num_of_relation_friend <= 0
							num_of_relation_lover <= 0
							add = 4
						}
						modifier = {
							stress_level >= 1
							add = {
								value = stress_level
								multiply = 2
							}
						}
						send_interface_message = {
							type = msg_health_change
							title = your_mind_withers_feed_message_title
							left_icon = root
							add_trait_xp = { trait = withering_mind track = withering_mind value = 50 }
						}
					}
					24 = {
						modifier = {
							num_of_relation_friend <= 0
							num_of_relation_lover <= 0
							num_of_relation_ward <= 0
							add = 12
						}
						modifier = {
							num_of_relation_friend <= 0
							num_of_relation_lover <= 0
							add = 12
						}
						modifier = {
							stress_level >= 1
							add = {
								value = stress_level
								multiply = 4
							}
						}
						send_interface_message = {
							type = msg_health_change
							title = your_mind_withers_feed_message_title
							left_icon = root
							add_trait_xp = { trait = withering_mind track = withering_mind value = 8 }
						}
					}
					75 = {
						send_interface_message = {
							type = msg_health_change
							title = your_mind_withers_feed_message_title
							left_icon = root
							add_trait_xp = { trait = withering_mind track = withering_mind value = 4 }
						}
					}
					0 = {
						modifier = {
							has_trait = family_first
							add = 10
						}
						modifier = {
							has_trait = seducer
							add = 10
						}
						modifier = {
							stress_level <= 0
							add = 10
						}
						modifier = {
							num_of_relation_friend >= 1
							add = {
								every_relation = {
									type = friend
									add = 25
								}
							}
						}
						modifier = {
							num_of_relation_lover >= 1
							add = {
								every_relation = {
									type = lover
									add = 15
								}
							}
						}
						modifier = {
							is_travelling = yes
							add = 50
						}
						modifier = {
							exists = involved_activity
							add = 50
						}
						send_interface_message = {
							type = msg_health_change
							title = your_mind_withers_slowly_feed_message_title
							left_icon = root
							add_trait_xp = { trait = withering_mind track = withering_mind value = 1 }
						}
					}
				}
				if = {
					limit = {
						has_trait_xp = { trait = withering_mind track = withering_mind value >= 100 }
					}
					add_trait = incapable
				}
			}
			if = {
				limit = {
					has_trait = fragile_bones
				}
				random_list = {
					25 = {
						send_interface_message = {
							type = msg_health_change
							title = your_bones_weaken_feed_message_title
							left_icon = root
							add_trait_xp = { trait = fragile_bones track = fragile_bones value = 3 }
						}
					}
					75 = {
						send_interface_message = {
							type = msg_health_change
							title = your_bones_weaken_feed_message_title
							left_icon = root
							add_trait_xp = { trait = fragile_bones track = fragile_bones value = 6 }
						}
					}
					0 = {
						modifier = {
							is_travelling = yes
							add = 100
						}
						modifier = {
							exists = involved_activity
							add = 50
						}
						modifier = {
							is_in_army = yes
							add = 100
						}
						send_interface_message = {
							type = msg_health_change
							title = your_bones_weaken_feed_message_title
							left_icon = root
							add_trait_xp = { trait = fragile_bones track = fragile_bones value = 12 }
						}
					}
				}
			}
			if = {
				limit = {
					has_trait = clouded_eyes
				}
				random_list = {
					5 = {
						modifier = { # Outside in the sun
							is_travelling = yes
							add = 15
						}
						modifier = { # Snow glare
							is_ruler = yes # For performance reasons we limit this as checking modifiers is expensive
							location_has_normal_or_worse_winter_trigger = yes
							add = 15
						}
						modifier = { # Snow glare
							location = {
								OR = {
									terrain = desert
									terrain = desert_mountains
								}
							}
							add = 10
						}
						send_interface_message = {
							type = msg_health_change
							title = your_vision_clouds_feed_message_title
							left_icon = root
							add_trait_xp = { trait = clouded_eyes track = clouded_eyes value = 15 }
						}
					}
					20 = {
						send_interface_message = {
							type = msg_health_change
							title = your_vision_clouds_feed_message_title
							left_icon = root
							add_trait_xp = { trait = clouded_eyes track = clouded_eyes value = 9 }
						}
					}
					75 = {
						send_interface_message = {
							type = msg_health_change
							title = your_vision_clouds_feed_message_title
							left_icon = root
							add_trait_xp = { trait = clouded_eyes track = clouded_eyes value = 3 }
						}
					}
				}
				if = {
					limit = {
						has_trait_xp = { trait = clouded_eyes track = clouded_eyes value >= 100 }
					}
					add_trait = blind
					remove_trait = clouded_eyes
				}
			}
		}
	}
}

wounded_recovery_pulse = {
	trigger = {
		OR = {
			has_trait = wounded
			has_infected_or_gangrenous_wound_trigger = yes
		}
	}
	random_events = {
		300 = 0
		400 = health.0001 #Recover from Wounded
		200 = health.0002 #Recover from Severely Wounded
		100 = health.0003 #Recover from Near Death
		150 = health.0004 #Infected wound is no longer infected
	}
}
