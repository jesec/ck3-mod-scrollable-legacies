steal_herd_success = {
    random_events = {
        100 = steal_herd_outcome.1001 # Success - 20%+ of their herd stolen
        5 = steal_herd_outcome.1003 # That was easy, maybe I can take something else?
    }
    effect = {
        scope:owner = {
            random = { # chance of inspiring a knight to become eligible for scoundrel accolade
                chance = 2
                modifier = {
                    factor = accolade_progress # scales with accolade progress
                }
                modifier = { # must have an eligible knight
                    NOT = {
                        any_knight = {
                            can_unlock_accolade_attribute_trigger = {
                                ATTRIBUTE = scoundrel
                            }
                        }
                    }
                    factor = 0
                }
                save_scope_as = accolade_liege
                set_variable = {
                    name = accolade_progress
                    value = 0
                }
                random_knight = {
                    limit = {
                        can_unlock_accolade_attribute_trigger = {
                            ATTRIBUTE = scoundrel
                        }
                    }
                    weight = {
                        base = 1
                        modifier = {
                            add = 50
                            is_acclaimed = yes
                        }
                    }
                    save_scope_as = accolade_knight
                }
                send_interface_message = {
                    type = msg_accolade_eligibility
                    title = accolade_scoundrel_unlock.t
                    left_icon = scope:accolade_knight
                    right_icon = scope:accolade_liege
                    custom_tooltip = accolade_scoundrel_unlock.tt
                    scope:accolade_knight = {
                        set_variable = {
                            name = scoundrel_attribute_unlock
                            value = yes
                        }
                    }   
                }
            }
        }
    }
}
