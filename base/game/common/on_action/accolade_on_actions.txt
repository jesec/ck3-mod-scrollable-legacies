# Root = Accolade
# Event targets: positive ( bool )
on_accolade_rank_change = {
	effect = {
		save_scope_as = changing_accolade
		acclaimed_knight = {
			save_scope_as = acclaimed_knight
		}
		accolade_owner = {
			# level up notification
			if = {
				limit = {
					scope:positive = yes
				}
				trigger_event = accolade.0001
			}
			# level down notification
			else = {
				trigger_event = accolade.0002
			}
		}
		
		if = {
			limit = {
				scope:changing_accolade = {
					accolade_rank >= 1
					acclaimed_knight = {
						exists = this
						NOT = {
							any_memory = {
								memory_type = accolade_ranked_up
							}
						}
					}
				}
				scope:positive = yes
			}
			scope:changing_accolade.acclaimed_knight = {
				create_character_memory = {
					type = accolade_ranked_up
					participants = {
						liege = root.accolade_owner
					}
				}
			}
		}
	}
}

# Root = Accolade
# Event targets: glory ( value )
on_accolade_glory_change = {
	effect = {
		if = {
			limit = {
				NOT = {
					has_variable = lifetime_glory
				}
			}
			set_variable = {
				name = lifetime_glory
				value = scope:glory
			}
		}
		else = {
			change_variable = {
				name = lifetime_glory
				add = scope:glory
			}
		}
	}
}

# Root = Accolade
on_accolade_created = {
	effect = {
		if = {
			limit = {
				acclaimed_knight ?= {
					NOT = {
						any_memory = {
							memory_type = accolade_created
						}
					}
				}
			}
			acclaimed_knight ?= {
				create_character_memory = {
					type = accolade_created
					participants = {
						liege = root.accolade_owner
					}
				}
			}
		}
	}
}

# Root = Accolade
# Event targets: old_acclaimed_knight, new_accolade_type ( yes/no )
on_accolade_acclaimed_death = {
	effect = {
		set_variable = {
			name = old_knight
			value = scope:old_acclaimed_knight
		}
		save_scope_as = succeeding_accolade
		accolade_owner = {
			accolade_knight_notification_with_glory_reset_effect = {
				REASON = death
			}
		}
	}
}

# Root = Accolade
# Event targets: old_acclaimed_knight, new_accolade_type ( yes/no ), new_owner
on_accolade_acclaimed_removal = {
	effect = {
		set_variable = {
			name = old_knight
			value = scope:old_acclaimed_knight
		}
		save_scope_as = succeeding_accolade
		#Invalidation event
		accolade_owner = {
			accolade_knight_notification_with_glory_reset_effect = {
				REASON = invalid
			}
		}
	}
}

# Root = Accolade
# Event targets: old_acclaimed_knight, new_accolade_type ( yes/no ), new_owner
on_accolade_create_squire = {
	effect = {
		#Used in the accolade_character_creation_effect scripted effect
		save_scope_as = succeeding_accolade
		#Used in the accolade_create_squire_effect scripted effect
		save_scope_as = accolade_in_need
		scope:old_acclaimed_knight ?= {
			save_scope_as = knight_in_need
		}
		scope:new_owner = {
			accolade_create_squire_effect = yes
		}
	}
}
