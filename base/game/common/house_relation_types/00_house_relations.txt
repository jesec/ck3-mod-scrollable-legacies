# House relations

default_house_relation = {
	neutral_level = neutral

	## Should these two ever have a relation?
	#
	# Various occurrences (e.g. event, marriage, declaration of war) might impact the relations between two houses.
	# If such a relation is not established then we must first create it in order for the occurrence to have something to impact.
	# Before creating a new relation we do however want to check if these two houses are relevant enough to track.
	#
	# This trigger verifies that a relation should indeed be created and kept track of.
	#
	# Scopes:
	#  house_1 (dynasty house) - One of the houses
	#  house_2 (dynasty house) – The other house
	#
	is_valid_to_start = {
		house_relation_is_valid_to_start_trigger = {
			HOUSE = scope:house_1
			OTHER_HOUSE = scope:house_2
		}
	}

	## Should we keep this relation around?
	#
	# Existing house relations are periodically pruned to to check if they should be kept, or removed.
	# Together with house relation decay (where a house that stays on 'neutral' for to long gets culled)
	# this keeps the  number of less relevant house relations down (thus saving some memory and CPU).
	#
	# Scopes:
	#  [root] (house relation) – The relation to evaluate the relevance of
	#  house_1 (dynasty house) - One of the houses (for convenience)
	#  house_2 (dynasty house) – The other house (for convenience)
	#
	is_valid_to_keep = {
		house_relation_is_valid_to_keep_trigger = {
			HOUSE = scope:house_1
			OTHER_HOUSE = scope:house_2
		}
	}

	levels = {
		feud = {
			opinion = -30
			cohesion_contribution = -30
			parameters = {
				may_use_eradicate_cb
				members_more_likely_to_scheme_hostile
				reduced_casus_belli_prestige_cost
				# From previous levels
				decreased_marriage_acceptance
				hostile_scheme_bonus
				decreased_activity_attendance
				more_likely_war_target
			}
		}
		rivalry = {
			opinion = -20
			cohesion_contribution = -20
			parameters = {
				hostile_scheme_bonus
				decreased_activity_attendance
				more_likely_war_target
				# From previous levels
				decreased_marriage_acceptance
			}
		}
		quarrel = {
			opinion = -10
			cohesion_contribution = -10
			parameters = {
				decreased_marriage_acceptance
			}
		}
		neutral = {
			cohesion_contribution = 0
			parameters = {}
		}
		cordial = {
			opinion = 10
			cohesion_contribution = 10
			parameters = {
				increased_marriage_acceptance
				increased_activity_attendance
			}
		}
		friendly = {
			opinion = 20
			cohesion_contribution = 20
			parameters = {
				personal_scheme_bonus
				less_likely_war_target
				# From previous levels
				increased_marriage_acceptance
				increased_activity_attendance
			}
		}
		amity = {
			opinion = 30
			cohesion_contribution = 30
			parameters = {
				unlocks_house_head_negotiate_alliance
				# From previous levels
				increased_marriage_acceptance
				increased_activity_attendance
				personal_scheme_bonus
				less_likely_war_target
			}
		}
	}
}
