secret_siphoned_treasury = {
	category = civil

	is_valid = {
		secret_siphoned_treasury_is_valid_trigger = {
			OWNER = scope:secret_owner
			TARGET = scope:secret_target
		}
	}

	is_shunned = {
		NOT = {
			secret_siphoned_treasury_is_criminal_trigger = {
				OWNER = scope:secret_owner
				TARGET = scope:secret_target
			}
		}
	}

	is_criminal = {
		secret_siphoned_treasury_is_criminal_trigger = {
			OWNER = scope:secret_owner
			TARGET = scope:secret_target
		}
	}

	on_discover = {}

	on_expose = {
		save_scope_as = secret
		#if you siphoned from yourself let the top liege get angry
		if = {
			limit = {
				scope:secret_target ?= scope:secret_owner
				scope:secret_target.liege != scope:secret_owner
			}
			scope:secret_target.liege = {
				save_scope_as = siphoned_treasury_victim
			}
		}
		#else let the discoverer get angry
		else_if = {
			limit = {
				scope:secret_target ?= scope:secret_owner
			}
			scope:discoverer = {
				save_scope_as = siphoned_treasury_victim
			}
		}
		#in any other case let the holder get angry
		else = {
			scope:secret_target ?= {
				save_scope_as = siphoned_treasury_victim
			}
		}
		# Grab the variable value in case it gets erased.
		save_scope_value_as = {
			name = embezzlement_stake
			value = var:embezzlement_stake
		}
		save_scope_value_as = {
			name = embezzlement_stake_half
			value = {
				value = var:embezzlement_stake
				multiply = 0.5
			}
		}
		
		scope:secret_owner = {
			save_scope_as = embezzler
			trigger_event = secrets.0121
		}
	}
}

secret_request_incursion = {
	category = raid

	is_valid = {
		secret_request_incursion_is_valid_trigger = {
			OWNER = scope:secret_owner
			TARGET = scope:secret_target
		}
	}

	is_shunned = {
		secret_request_incursion_is_criminal_trigger = no
	}

	is_criminal = {
		secret_request_incursion_is_criminal_trigger = yes
	}

	on_discover = {}

	on_expose = {
		save_scope_as = secret
		#if you requested incursion against yourself let the liege get angry
		if = {
			limit = {
				scope:secret_target ?= scope:secret_owner
				scope:secret_target.liege != scope:secret_owner
			}
			scope:secret_target.liege = {
				save_scope_as = secret_target
			}
		}
		#else let the discoverer get angry
		else_if = {
			limit = {
				scope:secret_target ?= scope:secret_owner
			}
			scope:secret_exposer = {
				save_scope_as = secret_target
			}
		}
		#in any other case let the holder get angry
		else = {
			scope:secret_target ?= {
				save_scope_as = secret_target
			}
		}
		if = {
			limit = {
				scope:secret_target = { is_alive = yes }
			}
			scope:secret_target = {
				# Apply appropriate criminal opinion.
				add_opinion = {
					target = scope:secret_owner
					modifier = requested_incursion_opinion
				}
			}
		}
		if = {
			limit = {
				scope:secret_target != scope:secret_owner.top_liege
				scope:secret_owner.top_liege != scope:secret_owner
			}
			scope:secret_owner.top_liege = {
				add_opinion = {
					target = scope:secret_owner
					modifier = requested_incursion_opinion
				}
			}
		}
		scope:secret_owner = {
			if = {
				limit = {
					exists = top_participant_group:dynastic_cycle
					title:h_china.holder ?= {
						this != scope:secret_owner
						this = scope:secret_target.top_liege
					}
				}
				tgp_activate_catalyst_against_hegemon_effect = {
					HEGEMON = scope:secret_owner.top_liege
					CATALYST = catalyst_requested_incursion
				}
			}
		}
		secret_exposed_notification_effect = yes
	}
}

secret_exam_cheater = {
	category = merit

	is_valid = {
		secret_exam_cheater_is_valid_trigger = {
			OWNER = scope:secret_owner
		}
	}

	is_shunned = {
		secret_exam_cheater_is_criminal_trigger = {
			OWNER = scope:secret_owner
		}
	}

	is_criminal = {
		secret_exam_cheater_is_criminal_trigger = {
			OWNER = scope:secret_owner
		}
	}

	on_discover = {}

	on_expose = {
		save_scope_as = secret
		scope:secret_owner = {
			if = {
				limit = {
					this != scope:secret_exposer
				}
				trigger_event = secrets.4001
			}
			else = {
				if = {
					limit = {
						has_character_modifier = tgp_passed_palace_exam_modifier
					}
					remove_character_flag = passed_palace_exam
					remove_character_modifier = tgp_passed_palace_exam_modifier
				}
				else = {
					remove_character_flag = passed_metropolitan_exam
					remove_character_modifier = tgp_passed_metropolitan_exam_modifier
				}
				force_step_down_landed_titles = yes
			}
		}
		secret_exposed_notification_effect = yes
	}
}
