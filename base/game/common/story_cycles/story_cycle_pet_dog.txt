story_cycle_pet_dog = {
	visible = yes

	icon = {
		trigger = {
			var:dog_fur_color ?= flag:black
			story_owner = {
				static_group_filter = {
					group = dog_ear_type
					match = 0.5
				}
			}
		}
		reference = "gfx/interface/icons/pets/artifact_dog_black.dds"
	}
	icon = {
		trigger = {
			var:dog_fur_color ?= flag:black
		}
		reference = "gfx/interface/icons/pets/artifact_dog_pointed_ears_black.dds"
	}
	icon = {
		trigger = {
			var:dog_fur_color ?= flag:gray
			story_owner = {
				static_group_filter = {
					group = dog_ear_type
					match = 0.5
				}
			}
		}
		reference = "gfx/interface/icons/pets/artifact_dog_pointed_ears_gray.dds"
	}
	icon = {
		trigger = {
			var:dog_fur_color ?= flag:gray
		}
		reference = "gfx/interface/icons/pets/artifact_dog_gray.dds"
	}
	icon = {
		trigger = {
			var:dog_fur_color ?= flag:brown
			story_owner = {
				static_group_filter = {
					group = dog_ear_type
					match = 0.5
				}
			}
		}
		reference = "gfx/interface/icons/pets/artifact_dog_brown.dds"
	}
	icon = {
		trigger = {
			var:dog_fur_color ?= flag:brown
		}
		reference = "gfx/interface/icons/pets/artifact_dog_pointed_ears_brown.dds"
	}
	icon = {
		trigger = {
			var:dog_fur_color ?= flag:white
			story_owner = {
				static_group_filter = {
					group = dog_ear_type
					match = 0.5
				}
			}
		}
		reference = "gfx/interface/icons/pets/artifact_dog_pointed_ears_white.dds"
	}
	icon = {
		trigger = {
			var:dog_fur_color ?= flag:white
		}
		reference = "gfx/interface/icons/pets/artifact_dog_white.dds"
	}
	icon = {
		trigger = {
			always = yes
		}
		reference = "gfx/interface/icons/pets/artifact_dog_folded_ears_fox_colored.dds"
	}
	background = {
		reference = "gfx/interface/illustrations/event_scenes/ep2_dog_kennels.dds"
	}
	
	visualization = {
		custom_string_key = "DOG_INFORMATION_STRING"
		modifiers = {
			dog_story_modifier
			dog_story_walking_modifier
			feast_well_fed_dog_modifier
			feral_pet_modifier
			dog_story_scenthound_modifier
			dog_story_beasthound_modifier
			adventurous_pet_modifier
			rat_hunting_dog_modifier
		}
		decisions = {
			pet_dog_decision
		}
	}
	
	on_setup = {
		if = {
			limit = {
				NOT = { exists = story_owner.var:dog_gender }
			}
			assign_dog_gender_effect = { GENDER = random }
		}
		if = {
			limit = {
				NOT = { exists = story_owner.var:dog_fur_color }
			}
			assign_dog_fur_color_effect = { COLOR = random }
		}
		if = {
			limit = {
				NOT = { exists = story_owner.var:dog_eye_color }
			}
			assign_dog_eye_color_effect = { COLOR = random }
		}
		set_variable = {
			name = dog_age_variable
			value = 0
		}


		story_owner = {
			add_character_modifier = {
				modifier = dog_story_modifier
			}
			add_character_flag = had_dog_story
		}
		
		# Give the dog a name!
		if = {
			limit = {
				story_owner = {
					NOT = { has_character_flag = is_naming_dog }
				}
				NOT = { exists = story_owner.var:story_cycle_dog_name }
			}
			story_owner = {
				add_character_flag = is_naming_dog
				trigger_event = {
					id = pet_animal.1001
					days = 2
				}
			}
		}
	}

	on_end = {
		story_owner = {
			remove_dog_story_modifiers_effect = yes
			remove_dog_name_effect = yes
		}
	}

	on_owner_death = {
		# Save info about the dog on primary heir for funeral events
		if = {
			limit = {
				exists = story_owner.player_heir
			}
			story_owner.player_heir = {
				set_variable = {
					name = dead_dog_owner
					years = 5
					value = root.story_owner
				}
				set_variable = {
					name = parent_dog_name
					years = 5
					value = root.var:story_cycle_dog_name
				}
				set_variable = {
					name = parent_dog_age
					years = 5
					value = root.var:dog_age_variable
				}
				set_variable = {
					name = parent_dog_fur_color
					years = 5
					value = root.var:dog_fur_color
				}
				set_variable = {
					name = parent_dog_eye_color
					years = 5
					value = root.var:dog_eye_color
				}
				set_variable = {
					name = parent_dog_gender
					years = 5
					value = root.var:dog_gender
				}
			}
		}
		scope:story = { end_story = yes }
	}

	#Age the dog
	effect_group = {
		days = 365

		trigger = {
			exists = var:dog_age_variable
		}

		triggered_effect = {
			trigger = { always = yes }
			effect = {
				change_variable = {
					name = dog_age_variable
					add = 1
				}
			}
		}
	}

	# The dog dies :(
	effect_group = {
		days = { 5000 7000 }
		chance = 100
	
		triggered_effect = {
			trigger = {
				exists = story_owner.var:story_cycle_dog_name
			}
			effect = {
				story_owner = {
					add_character_flag = {
						flag = dog_is_dying
					}
					trigger_event = pet_animal.1199
				}
			}
		}
	}

	# Something happens - random events
	effect_group = {
		days = { 365 600 }
		chance = 30

		trigger = {
			exists = story_owner.var:story_cycle_dog_name
			story_owner = {
				NOT = {
					has_character_flag = dog_is_dying
				}
			}
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					story_owner = {
						trigger_event = {
							on_action = ongoing_dog_events
						}
					}
				}
			}
		}
	}
}
