story_doppelganger = {
	# by James Beaumont
	visible = yes
	
	icon = {
		trigger = {
			always = yes
		}
		reference = "gfx/interface/icons/artifact/artifact_icons_four_tetrarchs.dds"
	}
	
	visualization = {
		custom_string_key = "DOPPELGANGER_STORY_PRACTICAL_INFORMATION_STRING"
		character = { variable_name = "doppelganger_character" label = "DOPPELGANGER" }
	}
	
	on_setup = {
		
	}
#
	on_end = {
		debug_log = "Doppelganger story ended on:"
		debug_log_date = yes
	}
#
	on_owner_death = {
		var:doppelganger_character = {
			if = {
				limit = { is_alive = yes }
				silent_disappearance_effect = yes
			}
		}
	}

	# Doppelganger dies, end the story
	effect_group = {
		days = { 40 60 }

		trigger = {
			exists = var:doppelganger_character
		}

		first_valid = {
			triggered_effect = {
				trigger = {
					exists = var:doppelganger_character
					var:doppelganger_character = {
						is_alive = no
					}
				}
				effect = {
					end_story = yes
				}
			}
		}
	}

	# Something happens - random events
	effect_group = {
		days = { 365 600 }
		chance = 50

		trigger = {
			exists = var:doppelganger_character
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					story_owner = {
						trigger_event = {
							on_action = ongoing_doppelganger_events
						}
					}
				}
			}
		}
	}

	#Maintenance group
	effect_group = {
		days = 1
		trigger = { 
			story_owner = { has_government = landless_adventurer_government }
		}
		triggered_effect = {
			trigger = { always = yes }
			effect = {
				#You became a landless adventurer! Ending story prematurely.
				story_owner = { save_scope_as = story_owner }
				debug_log = "Doppelganger invalidated"
				debug_log_scopes = yes
				end_story = yes
			}
		}
	}
}
