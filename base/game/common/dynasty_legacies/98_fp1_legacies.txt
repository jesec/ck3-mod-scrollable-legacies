fp1_adventure_legacy_track = {
	is_shown = {
		has_dlc_feature = the_northern_lords
		OR = {
			has_game_rule = unrestricted_dynasty_legacies_all
			AND = {
				has_game_rule = unrestricted_dynasty_legacies_player_only
				is_ai = no
			}
			dynasty = {
				OR = {
					dynast = {
						culture = {
							OR = {
								this = culture:norse
								any_parent_culture_or_above = {
									this = culture:norse
								}
							}
						}
					}
					has_dynasty_perk = fp1_adventure_legacy_1
				}
			}
		}
	}
}
